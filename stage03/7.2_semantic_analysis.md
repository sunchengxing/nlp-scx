## 1 项目背景与目标

### 应用场景

消防执法人员在现场巡查时产生大量语音对话录音（巡查记录、问询笔录、整改通知等），
人工整理耗时且容易遗漏关键违规信息。

### 目标

| 输入 | 输出 |
|------|------|
| 消防现场语音对话（数分钟录音） | 结构化违规报告 |
| - | 对应消防法规条款 |
| - | 整改建议 |

### 核心 NLP 子任务

```
语音对话
  ├── 语义压缩       → 提炼核心事实
  ├── 违规行为分类   → 归属违规类别
  ├── 命名实体识别   → 抽取位置/设施/行为
  └── 法规检索       → 匹配相关条款
```

### 4.2 语音转文字（asr.py）

```python
import whisper

def transcribe(audio_path: str, model_size: str = "medium") -> str:
    """
    使用 Whisper 将音频转为文字。
    model_size: tiny/base/small/medium/large，中文推荐 medium 以上
    """
    model = whisper.load_model(model_size)
    result = model.transcribe(audio_path, language="zh")
    return result["text"]

if __name__ == "__main__":
    text = transcribe("sample.wav")
    print(text)
```

**说明：**
- 首次运行自动下载模型（约 1.5GB for medium）
- 离线使用，无需 API Key
- 消防对话中专有名词较多，建议用 `medium` 或 `large`


### 4.3 LLM 语义压缩与结构化抽取（extractor.py）

```python
from openai import OpenAI

client = OpenAI(
    api_key="your-api-key",
    base_url="https://dashscope.aliyuncs.com/compatible-mode/v1"  # Qwen
)

SYSTEM_PROMPT = """你是消防安全专家。分析消防巡查对话，输出 JSON 格式：
{
  "summary": "一句话概括核心事实",
  "violations": [
    {
      "location": "具体位置",
      "facility": "涉及设施",
      "behavior": "违规行为描述",
      "category": "违规类别（疏散通道/消防设施/用火用电/防火分区/其他）",
      "severity": "严重/一般/轻微"
    }
  ],
  "key_facts": ["事实1", "事实2"]
}
只输出 JSON，不要其他内容。"""

def extract_violations(dialogue: str) -> dict:
    response = client.chat.completions.create(
        model="qwen-plus",
        messages=[
            {"role": "system", "content": SYSTEM_PROMPT},
            {"role": "user", "content": f"以下是消防巡查对话：\n{dialogue}"}
        ],
        temperature=0.1
    )
    import json
    return json.loads(response.choices[0].message.content)

if __name__ == "__main__":
    sample = """
    检查员：这里走廊里放了这么多纸箱，疏散通道都堵了。
    负责人：哦这是仓库临时放的，我让他们搬。
    检查员：还有这个灭火器，看铭牌是2021年的，早过期了。
    负责人：这个我们忘记检查了。
    """
    result = extract_violations(sample)
    import json
    print(json.dumps(result, ensure_ascii=False, indent=2))
```

**Prompt 设计要点：**
- 明确角色（消防专家）避免通用回答
- 约束输出格式为 JSON，便于后续处理
- `temperature=0.1` 降低随机性，保证结构稳定
- 若对话过长（>3000字）先分段压缩再合并

### 5.2 构建法规向量库

```python
import json
import faiss
import numpy as np
from sentence_transformers import SentenceTransformer

EMBED_MODEL = "shibing624/text2vec-base-chinese"

def build_regulation_index(reg_path: str, index_path: str = "reg.index"):
    """将法规库向量化并保存 Faiss 索引。"""
    model = SentenceTransformer(EMBED_MODEL)
    with open(reg_path, encoding="utf-8") as f:
        regs = json.load(f)

    texts = [r["title"] + " " + r["content"] for r in regs]
    embeddings = model.encode(texts, normalize_embeddings=True)

    dim = embeddings.shape[1]
    index = faiss.IndexFlatIP(dim)  # 内积=余弦相似度（已归一化）
    index.add(embeddings.astype(np.float32))

    faiss.write_index(index, index_path)
    with open(index_path + ".meta.json", "w", encoding="utf-8") as f:
        json.dump(regs, f, ensure_ascii=False, indent=2)
    print(f"索引已保存：{index_path}，共 {len(regs)} 条法规")
```

### 5.3 违规行为检索匹配（retriever.py）

```python
import faiss
import json
import numpy as np
from sentence_transformers import SentenceTransformer

EMBED_MODEL = "shibing624/text2vec-base-chinese"

class RegulationRetriever:
    def __init__(self, index_path: str = "reg.index"):
        self.model = SentenceTransformer(EMBED_MODEL)
        self.index = faiss.read_index(index_path)
        with open(index_path + ".meta.json", encoding="utf-8") as f:
            self.regulations = json.load(f)

    def search(self, query: str, top_k: int = 3) -> list[dict]:
        """输入违规行为描述，返回 Top-K 相关法规。"""
        vec = self.model.encode([query], normalize_embeddings=True)
        scores, indices = self.index.search(vec.astype(np.float32), top_k)
        results = []
        for score, idx in zip(scores[0], indices[0]):
            reg = self.regulations[idx].copy()
            reg["similarity"] = float(score)
            results.append(reg)
        return results

if __name__ == "__main__":
    r = RegulationRetriever()
    hits = r.search("走廊堆放杂物，疏散通道被堵")
    for h in hits:
        print(f"[{h['similarity']:.3f}] {h['title']}")
        print(f"  内容：{h['content']}")
        print(f"  处罚：{h['penalty']}\n")
```

| 编号 | 类别 |
|------|------|

### 6.2 训练代码（classifier.py）

```python
import json
import torch
from torch.utils.data import Dataset, DataLoader
from transformers import (
    BertTokenizer, BertForSequenceClassification,
    AdamW, get_linear_schedule_with_warmup
)

BERT_MODEL = "hfl/chinese-roberta-wwm-ext"
LABELS = ["疏散通道堵塞", "消防设施缺失", "违规用火用电", "防火分区破坏", "其他"]
NUM_LABELS = len(LABELS)

class ViolationDataset(Dataset):
    def __init__(self, data_path, tokenizer, max_len=128):
        with open(data_path, encoding="utf-8") as f:
            self.data = json.load(f)
        self.tokenizer = tokenizer
        self.max_len = max_len

    def __len__(self):
        return len(self.data)

    def __getitem__(self, idx):
        item = self.data[idx]
        enc = self.tokenizer(
            item["text"], max_length=self.max_len,
            padding="max_length", truncation=True, return_tensors="pt"
        )
        return {
            "input_ids": enc["input_ids"].squeeze(),
            "attention_mask": enc["attention_mask"].squeeze(),
            "label": torch.tensor(item["label"], dtype=torch.long)
        }

def train(train_path: str, save_dir: str = "models/cls", epochs: int = 5):
    tokenizer = BertTokenizer.from_pretrained(BERT_MODEL)
    model = BertForSequenceClassification.from_pretrained(
        BERT_MODEL, num_labels=NUM_LABELS
    )
    dataset = ViolationDataset(train_path, tokenizer)
    loader = DataLoader(dataset, batch_size=16, shuffle=True)

    optimizer = AdamW(model.parameters(), lr=2e-5)
    scheduler = get_linear_schedule_with_warmup(
        optimizer, num_warmup_steps=50,
        num_training_steps=len(loader) * epochs
    )
    model.train()
    for epoch in range(epochs):
        total_loss = 0
        for batch in loader:
            optimizer.zero_grad()
            out = model(
                input_ids=batch["input_ids"],
                attention_mask=batch["attention_mask"],
                labels=batch["label"]
            )
            out.loss.backward()
            optimizer.step()
            scheduler.step()
            total_loss += out.loss.item()
        print(f"Epoch {epoch+1}/{epochs}  loss={total_loss/len(loader):.4f}")

    model.save_pretrained(save_dir)
    tokenizer.save_pretrained(save_dir)
    print(f"模型已保存至 {save_dir}")

def predict(text: str, model_dir: str = "models/cls") -> str:
    tokenizer = BertTokenizer.from_pretrained(model_dir)
    model = BertForSequenceClassification.from_pretrained(model_dir)
    model.eval()
    enc = tokenizer(text, return_tensors="pt", truncation=True, max_length=128)
    with torch.no_grad():
        logits = model(**enc).logits
    label_id = logits.argmax(-1).item()
    return LABELS[label_id]
```

### 7.1 标注格式（BIO 标注法）

```

```

**实体类型：**

| 标签 | 含义 | 示例 |
|------|------|------|
| LOC | 位置 | 三楼走廊、东侧安全出口 |
| FAC | 设施 | 防火门、灭火器、消防栓 |
| VIO | 违规行为 | 被堵塞、已损坏、超期未换 |

**标注工具推荐：** Label Studio（开源，支持 NER 序列标注）


### 7.2 BERT + CRF 模型代码（ner.py）

```python
import torch
import torch.nn as nn
from transformers import BertModel, BertTokenizerFast
from torchcrf import CRF

BERT_MODEL = "hfl/chinese-roberta-wwm-ext"
TAG2ID = {"O":0, "B-LOC":1, "I-LOC":2, "B-FAC":3, "I-FAC":4, "B-VIO":5, "I-VIO":6}
ID2TAG = {v: k for k, v in TAG2ID.items()}

class BertCRF(nn.Module):
    def __init__(self, num_tags: int):
        super().__init__()
        self.bert = BertModel.from_pretrained(BERT_MODEL)
        self.dropout = nn.Dropout(0.1)
        self.fc = nn.Linear(self.bert.config.hidden_size, num_tags)
        self.crf = CRF(num_tags, batch_first=True)

    def forward(self, input_ids, attention_mask, labels=None):
        out = self.bert(input_ids=input_ids, attention_mask=attention_mask)
        emissions = self.fc(self.dropout(out.last_hidden_state))
        if labels is not None:
            mask = attention_mask.bool()
            loss = -self.crf(emissions, labels, mask=mask, reduction="mean")
            return loss
        return self.crf.decode(emissions, mask=attention_mask.bool())

def extract_entities(text: str, model: BertCRF, tokenizer) -> list[dict]:
    """推理：从文本中抽取实体。"""
    enc = tokenizer(list(text), is_split_into_words=True,
                    return_tensors="pt", truncation=True, max_length=128)
    with torch.no_grad():
        tag_ids = model(enc["input_ids"], enc["attention_mask"])[0]

    tokens = list(text)
    entities, current = [], None
    for token, tag_id in zip(tokens, tag_ids[1:-1]):  # 去掉[CLS][SEP]
        tag = ID2TAG.get(tag_id, "O")
        if tag.startswith("B-"):
            if current:
                entities.append(current)
            current = {"type": tag[2:], "text": token}
        elif tag.startswith("I-") and current and current["type"] == tag[2:]:
            current["text"] += token
        else:
            if current:
                entities.append(current)
            current = None
    if current:
        entities.append(current)
    return entities
```

## 8 完整 Pipeline 整合（pipeline.py）

```python
import json
from asr import transcribe
from extractor import extract_violations
from retriever import RegulationRetriever
from openai import OpenAI

client = OpenAI(
    api_key="your-api-key",
    base_url="https://dashscope.aliyuncs.com/compatible-mode/v1"
)
retriever = RegulationRetriever("reg.index")

REPORT_PROMPT = """根据以下消防违规信息和相关法规，生成一份专业的违规处理报告。
要求：
1. 逐条列出违规行为
2. 注明对应法规条款和编号
3. 给出具体整改建议和时限
4. 说明不整改的法律后果

违规信息：
{violations}

相关法规：
{regulations}
"""

def run_pipeline(audio_path: str) -> dict:
    print("Step 1: 语音转文字...")
    dialogue = transcribe(audio_path)

    print("Step 2: LLM语义抽取...")
    extracted = extract_violations(dialogue)

    print("Step 3: 法规检索...")
    all_regulations = []
    for v in extracted.get("violations", []):
        query = v["location"] + v["behavior"]
        hits = retriever.search(query, top_k=2)
        for h in hits:
            v.setdefault("regulations", []).append({
                "title": h["title"],
                "content": h["content"],
                "penalty": h["penalty"]
            })
            all_regulations.append(h)

    print("Step 4: 生成报告...")
    reg_text = "\n".join(
        f"- {r['title']}: {r['content']}  处罚：{r['penalty']}"
        for r in all_regulations
    )
    vio_text = json.dumps(extracted["violations"], ensure_ascii=False, indent=2)
    resp = client.chat.completions.create(
        model="qwen-plus",
        messages=[{
            "role": "user",
            "content": REPORT_PROMPT.format(
                violations=vio_text, regulations=reg_text
            )
        }],
        temperature=0.3
    )
    report = resp.choices[0].message.content

    return {
        "original_dialogue": dialogue,
        "extracted": extracted,
        "report": report
    }

if __name__ == "__main__":
    result = run_pipeline("sample.wav")
    print("===== 违规报告 =====")
    print(result["report"])
    with open("output_report.json", "w", encoding="utf-8") as f:
        json.dump(result, f, ensure_ascii=False, indent=2)
    print("\n完整结果已保存至 output_report.json")
```

```
对应法规：
《建筑设计防火规范》GB50016-2018 第7.3条
整改建议：
- 72小时内更换同型号有效期内灭火器
- 建立灭火器定期检查台账（每半年一次）
```
